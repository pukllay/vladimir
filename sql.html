       DECLARE @personal TABLE(
        nombre VARCHAR(50),
        salario DECIMAL(18,2)
    );
    INSERT INTO @personal VALUES ('Daly', 3500.50),('Daniela', 2500.50),('Karla', 1500.00);
    SELECT * FROM @personal;
    
    WITH salarioPromedio AS (
        SELECT AVG(salario) AS promedio
        FROM @personal
    )
    SELECT 
        nombre, salario
    FROM 
        @personal
    WHERE 
        salario > (SELECT promedio FROM salarioPromedio);
